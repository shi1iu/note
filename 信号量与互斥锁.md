1.消费者-生产者

```C
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <semaphore.h>
#include <unistd.h>

int buffer = 0; // 共享缓冲区

pthread_mutex_t mutex;   // 互斥锁：保护 buffer
sem_t full;              // 表示缓冲区“满”的信号量（初始0）
sem_t empty;             // 表示缓冲区“空”的信号量（初始1）

void* producer(void* arg) {
    for (int i = 1; i <= 5; i++) {
        sem_wait(&empty);              // 等待缓冲区为空
        pthread_mutex_lock(&mutex);    // 加锁保护共享数据

        buffer = i;
        printf("生产者生产了：%d\n", buffer);

        pthread_mutex_unlock(&mutex);  // 解锁
        sem_post(&full);               // 通知消费者“已生产”
        sleep(1); // 模拟耗时
    }
    return NULL;
}

void* consumer(void* arg) {
    for (int i = 1; i <= 5; i++) {
        sem_wait(&full);               // 等待缓冲区有数据
        pthread_mutex_lock(&mutex);    // 加锁保护共享数据

        printf("消费者消费了：%d\n", buffer);
        buffer = 0;

        pthread_mutex_unlock(&mutex);  // 解锁
        sem_post(&empty);              // 通知生产者“已消费”
        sleep(2); // 模拟耗时
    }
    return NULL;
}

int main() {
    pthread_t prod, cons;

    pthread_mutex_init(&mutex, NULL);
    sem_init(&full, 0, 0);   // 初始化为0，表示开始时没有产品
    sem_init(&empty, 0, 1);  // 初始化为1，表示缓冲区空着

    pthread_create(&prod, NULL, producer, NULL);
    pthread_create(&cons, NULL, consumer, NULL);

    pthread_join(prod, NULL);
    pthread_join(cons, NULL);

    pthread_mutex_destroy(&mutex);
    sem_destroy(&full);
    sem_destroy(&empty);

    return 0;
}
```
